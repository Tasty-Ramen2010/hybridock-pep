5. _contact_zone_indices(pose_ca, receptor_ca, cutoff=6.0) -> np.ndarray:
   Numpy broadcasting: diff = pose_ca[:, np.newaxis, :] - receptor_ca[np.newaxis, :, :]
   dists = np.sqrt(np.sum(diff ** 2, axis=2))  # shape [n_pep, n_rec]
   return np.where(np.any(dists < cutoff, axis=1))[0]
   NOTE: This function returns the raw contact indices. The D-02 fallback (< 3 → full peptide)
   is applied in _build_rmsd_matrix at the pair level using intersection logic.

6. _build_rmsd_matrix(ca_arrays: list[np.ndarray], contact_indices: list[np.ndarray]) -> np.ndarray:
   Explicit O(n²) loop (not pairwise_distances callable — see RESEARCH.md Pattern 3 note).
   For each pair (i, j): common = np.intersect1d(contact_indices[i], contact_indices[j])
   if len(common) < 3: common = np.arange(min(len(ca_arrays[i]), len(ca_arrays[j])))
   coords_i = ca_arrays[i][common]; coords_j = ca_arrays[j][common]
   rmsd = float(np.sqrt(np.mean(np.sum((coords_i - coords_j) ** 2, axis=1))))
   dist[i, j] = dist[j, i] = rmsd

7. _select_k_silhouette(dist_matrix: np.ndarray) -> tuple[int, dict[int, float]]:
   k_max = min(15, dist_matrix.shape[0] // 5)
   if k_max < 2: return 2, {}
   sil_scores: dict[int, float] = {}
   for k in range(2, k_max + 1):
       agg = AgglomerativeClustering(n_clusters=k, metric="precomputed", linkage="average")
       labels = agg.fit_predict(dist_matrix)
       try:
           score = silhouette_score(dist_matrix, labels, metric="precomputed")
           sil_scores[k] = float(score)
       except ValueError:
           logger.debug("silhouette_score failed for k=%d, skipping", k)
           continue
   if not sil_scores: return 2, {}
   k_optimal = max(sil_scores, key=lambda k: sil_scores[k])
   return k_optimal, sil_scores

8. cluster_poses(scored_poses: list[ScoredPose], config: DockConfig) -> ClusterResult:
   - Log "Stage 3: clustering %d poses", len(scored_poses)
   - receptor_ca = _load_receptor_ca_coords(config.receptor_path.resolve())
   - contact_indices = [_contact_zone_indices(p.ca_coords, receptor_ca) for p in scored_poses]
   - ca_arrays = [p.ca_coords for p in scored_poses]
   - dist_matrix = _build_rmsd_matrix(ca_arrays, contact_indices)
   - k_optimal, sil_scores = _select_k_silhouette(dist_matrix)
   - Final fit: agg = AgglomerativeClustering(n_clusters=k_optimal, metric="precomputed", linkage="average")
   - labels = agg.fit_predict(dist_matrix)
   - Mutate in-place: for i, pose in enumerate(scored_poses): pose.cluster_id = int(labels[i])
   - Log "Assigned %d poses to %d clusters", len(scored_poses), k_optimal
   - Lazy import stats + plotting (defer sklearn dependency):
     from hybridock_pep.analysis.statistics import compute_cluster_stats, write_cluster_summary_csv
     from hybridock_pep.analysis.plotting import plot_convergence, plot_silhouette
   - stats = compute_cluster_stats(scored_poses)
   - write_cluster_summary_csv(stats, config.output_dir / "cluster_summary.csv")
   - plot_convergence(scored_poses, config.output_dir / "convergence_plot.png")
   - sil_val = sil_scores.get(k_optimal, float("nan"))

   - return ClusterResult(k_optimal=k_optimal, silhouette_score=sil_val, per_cluster_stats=stats)
