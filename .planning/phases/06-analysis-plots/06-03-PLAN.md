3. _ci95(values: list[float]) -> tuple[float, float]:
   ```
   n = len(values)
   if n < 2:
       mean = float(values[0]) if values else float("nan")
       return mean, mean
   arr = np.asarray(values, dtype=np.float64)
   mean = float(arr.mean())
   sem = float(arr.std(ddof=1) / np.sqrt(n))   # standard error of mean, NOT std
   lo, hi = t_dist.interval(0.95, df=n - 1, loc=mean, scale=sem)
   return float(lo), float(hi)
   ```
   If t_dist is None (ImportError guard): fall back to returning (mean, mean) with a warning log.
   CRITICAL: scale=sem (std/sqrt(n)), NOT scale=std. df=n-1, NOT df=n.

4. compute_cluster_stats(scored_poses: list[ScoredPose]) -> list[dict[str, Any]]:
   Group poses by cluster_id. For each cluster (sorted ascending by cluster_id):
   - scores = [p.hybrid_score for p in cluster_poses]  (all are float at this stage)
   - n = len(scores)
   - mean_hybrid = float(np.mean(scores))
   - std_hybrid = float(np.std(scores, ddof=1)) if n > 1 else 0.0
   - ci95_lower, ci95_upper = _ci95(scores)
   - best_pose_idx = min(cluster_poses, key=lambda p: p.hybrid_score).pose_idx
   Return list sorted by cluster_id ascending.
   Log: logger.debug("Cluster %d: n=%d, mean=%.3f, sil N/A", cluster_id, n, mean_hybrid)

5. write_cluster_summary_csv(stats: list[dict[str, Any]], output_path: Path) -> None:
   FIELDNAMES = ["cluster_id", "n_poses", "mean_hybrid_score", "std_hybrid_score",
                 "ci95_lower", "ci95_upper", "best_pose_idx"]
   output_path.parent.mkdir(parents=True, exist_ok=True)
   with output_path.open("w", newline="") as fh:
       writer = csv.DictWriter(fh, fieldnames=FIELDNAMES)
       writer.writeheader()
       writer.writerows(stats)
   logger.info("Wrote cluster_summary.csv to %s", output_path)
