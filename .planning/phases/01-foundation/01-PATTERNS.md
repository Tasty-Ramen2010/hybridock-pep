**DockConfig — Pydantic v2 BaseModel pattern:**
```python
from __future__ import annotations

from pathlib import Path
from typing import Literal

from pydantic import BaseModel, ConfigDict, field_validator, model_validator


_VALID_AA = frozenset("ACDEFGHIKLMNPQRSTVWY")


class DockConfig(BaseModel):
    model_config = ConfigDict(frozen=True)   # Claude's discretion: frozen preferred (immutable after construction)

    peptide_sequence: str
    receptor_path: Path
    site_coords: tuple[float, float, float]
    box_size: float
    n_samples: int = 100
    seed: int | None = None
    scoring: set[Literal["vina", "ad4"]] = {"vina", "ad4"}
    output_dir: Path
    run_id: str                              # generated by model_validator (see below)
    verbosity: int = 0

    @field_validator("peptide_sequence")
    @classmethod
    def _validate_peptide(cls, v: str) -> str:
        bad = set(v.upper()) - _VALID_AA
        if bad:
            raise ValueError(f"Non-standard amino acid characters: {bad}")
        return v.upper()

    @field_validator("receptor_path", mode="after")
    @classmethod
    def _receptor_exists(cls, v: Path) -> Path:
        if not v.exists():
            raise ValueError(f"Receptor path does not exist: {v}")
        return v

    @model_validator(mode="before")
    @classmethod
    def _generate_run_id(cls, data: dict) -> dict:
        if "run_id" not in data:
            import hashlib, time
            seed_part = hashlib.sha1(str(data.get("seed")).encode()).hexdigest()[:8]
            data["run_id"] = f"{int(time.time())}_{seed_part}"
        return data
```

Key Pydantic v2 specifics:
- `@field_validator` replaces v1's `@validator`. Always `@classmethod`.
- `mode="after"` receives the already-coerced value (Path object). `mode="before"` receives raw input.
- `@model_validator(mode="before")` runs before field validation — correct place to inject `run_id`.
- `ConfigDict(frozen=True)` makes the model hashable and immutable. Correct for a config object passed across subprocess boundaries.

**PoseRecord — plain dataclass pattern:**
```python
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path

import numpy as np


@dataclass
class PoseRecord:
    """Parsed peptide pose with pre-extracted Cα coordinates.

    Args:
        pose_idx: Zero-based index of this pose within the sampling run.
        pdb_path: Absolute path to the raw PDB file produced by RAPiDock.
        sequence: Single-letter amino acid sequence parsed from the PDB.
        ca_coords: Shape (n_residues, 3) float64 array of Cα XYZ coordinates.
            Populated at parse time; never re-read from disk.
    """
    pose_idx: int
    pdb_path: Path
    sequence: str
    ca_coords: np.ndarray   # shape (n_residues, 3), dtype float64
```

**ScoredPose — dataclass inheritance pattern (D-07):**
```python
@dataclass
class ScoredPose(PoseRecord):
    """PoseRecord extended with scoring results.

    All score fields default to None; they are filled in sequentially
    by the scoring pipeline. hybrid_score is set last.
    """
    vina_score: float | None = None
    ad4_score: float | None = None
    entropy_correction: float | None = None
    hybrid_score: float | None = None
    cluster_id: int | None = None
    pdbqt_path: Path | None = None
    is_ad4_anomaly: bool = False   # True when ad4_score > 0 (SCORE-02)
    is_clipped: bool = False       # True when atoms were outside grid bounds (SCORE-01)
```

Important: `@dataclass` inheritance requires that parent fields have no defaults if child
adds fields without defaults. `PoseRecord` has no default fields, so `ScoredPose` can add
defaulted fields freely. This is correct.

### `src/hybridock_pep/cli.py` (CLI stub, request-response)

**Analog:** None — argparse subcommand pattern from stdlib.

Phase 1 creates a stub; full subcommand logic is Phase 2+. The stub must:
1. Define the `main()` entry point referenced by `pyproject.toml`
2. Register subcommand names (`dock`, `calibrate`, `benchmark`, `prep`) so the entry
   point resolves without error
3. Print help on bare invocation

```python
from __future__ import annotations

import argparse
import logging

logger = logging.getLogger(__name__)


def _build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="hybridock-pep",
        description="Hybrid peptide docking: RAPiDock sampling + physics-based rescoring.",
    )
    parser.add_argument("-v", "--verbose", action="count", default=0,
                        help="Increase verbosity (-v INFO, -vv DEBUG).")
    sub = parser.add_subparsers(dest="command", metavar="COMMAND")
    sub.required = False

    # Stubs — populated in later phases
    sub.add_parser("dock", help="Run end-to-end docking pipeline.")
    sub.add_parser("calibrate", help="Calibrate entropy correction coefficient alpha.")
    sub.add_parser("benchmark", help="Run benchmark suite against reference complexes.")
    sub.add_parser("prep", help="Prepare receptor/ligand files without running docking.")

    return parser


def main() -> None:
    parser = _build_parser()
    args = parser.parse_args()

    log_level = logging.WARNING if args.verbose == 0 else (
        logging.INFO if args.verbose == 1 else logging.DEBUG
    )
    logging.basicConfig(level=log_level, format="%(levelname)s %(name)s: %(message)s")

    if args.command is None:
        parser.print_help()
        return

    # Phase 2+ will dispatch here
    parser.error(f"Subcommand '{args.command}' is not yet implemented.")
```

Note: CLAUDE.md §4 specifies `logging` module (not `print`) and INFO by default, DEBUG on
`-v`. The `-v` flag above maps `0→WARNING` at the stub stage (quieter for a stub), but the
planner may prefer `0→INFO` to match §4 exactly.

### `scripts/smoke_test.sh` (script/utility, request-response)

**Analog:** None — bash script pattern with portable POSIX checks.

D-13 specifies three checks: CUDA CC >= 12.0, ADFRsuite `prepare_receptor4.py` on PATH,
Vina >= 1.2.5. On macOS ARM (no CUDA) the CUDA check warns but does not fail.

Pattern structure:
```bash
#!/usr/bin/env bash
# scripts/smoke_test.sh — validate environment dependencies for HybriDock-Pep
set -euo pipefail

PASS=0; WARN=0; FAIL=0

pass()  { echo "[PASS] $*"; ((PASS++));  }
warn()  { echo "[WARN] $*"; ((WARN++));  }
fail()  { echo "[FAIL] $*"; ((FAIL++));  }

# --- CUDA compute capability check ---
# Skip on macOS ARM (no nvidia-smi) with a warning
if command -v nvidia-smi &>/dev/null; then
    CC=$(nvidia-smi --query-gpu=compute_cap --format=csv,noheader | head -1)
    MAJOR=$(echo "$CC" | cut -d. -f1)
    MINOR=$(echo "$CC" | cut -d. -f2)
    if [ "$MAJOR" -gt 12 ] || { [ "$MAJOR" -eq 12 ] && [ "$MINOR" -ge 0 ]; }; then
        pass "CUDA compute capability $CC >= 12.0"
    else
        fail "CUDA compute capability $CC < 12.0 — Blackwell (RTX 5070) requires CC 12.0+"
    fi
else
    warn "nvidia-smi not found — skipping CUDA check (expected on macOS ARM)"
fi

# --- ADFRsuite check ---
if command -v prepare_receptor4.py &>/dev/null; then
    pass "prepare_receptor4.py found on PATH"
else
    fail "prepare_receptor4.py not on PATH — install ADFRsuite from https://ccsb.scripps.edu/adfrsuite/"
fi

# --- Vina version check ---
if command -v vina &>/dev/null; then
    VINA_VER=$(vina --version 2>&1 | grep -oE '[0-9]+\.[0-9]+\.[0-9]+' | head -1)
    # ... semver comparison
    pass "AutoDock Vina $VINA_VER"
else
    fail "vina not on PATH — install via: pip install vina>=1.2.5"
fi

# --- Summary ---
echo ""
echo "Results: ${PASS} passed, ${WARN} warnings, ${FAIL} failed"
[ "$FAIL" -eq 0 ] || exit 1
```

The script must be committed as executable (`chmod +x scripts/smoke_test.sh`).

### `tests/test_models.py` (test)

**Analog:** None — pytest pattern.

Phase 1 test scope: validate that `DockConfig`, `PoseRecord`, `ScoredPose`, `PoseFailure`
construct correctly and that `DockConfig` raises `ValidationError` on bad inputs.

```python
from __future__ import annotations

import numpy as np
import pytest
from pathlib import Path
from pydantic import ValidationError

from hybridock_pep import DockConfig, PoseFailure, PoseRecord, ScoredPose


@pytest.fixture()
def valid_receptor(tmp_path: Path) -> Path:
    p = tmp_path / "receptor.pdb"
    p.write_text("ATOM      1  N   ALA A   1       0.000   0.000   0.000  1.00  0.00           N\n")
    return p


class TestDockConfig:
    def test_valid_construction(self, valid_receptor: Path, tmp_path: Path) -> None:
        cfg = DockConfig(
            peptide_sequence="LISDAELEAIFEADC",
            receptor_path=valid_receptor,
            site_coords=(22.5, 14.1, 38.7),
            box_size=20.0,
            output_dir=tmp_path,
        )
        assert cfg.peptide_sequence == "LISDAELEAIFEADC"
        assert isinstance(cfg.run_id, str)
        assert len(cfg.run_id) > 0

    def test_invalid_peptide_sequence(self, valid_receptor: Path, tmp_path: Path) -> None:
        with pytest.raises(ValidationError, match="Non-standard amino acid"):
            DockConfig(
                peptide_sequence="LISB",   # B is not a standard AA
                receptor_path=valid_receptor,
                site_coords=(0.0, 0.0, 0.0),
                box_size=20.0,
                output_dir=tmp_path,
            )

    def test_missing_receptor(self, tmp_path: Path) -> None:
        with pytest.raises(ValidationError, match="does not exist"):
            DockConfig(
                peptide_sequence="LIS",
                receptor_path=tmp_path / "nonexistent.pdb",
                site_coords=(0.0, 0.0, 0.0),
                box_size=20.0,
                output_dir=tmp_path,
            )


class TestPoseRecord:
    def test_construction(self, tmp_path: Path) -> None:
        ca = np.zeros((15, 3), dtype=np.float64)
        pr = PoseRecord(pose_idx=0, pdb_path=tmp_path / "pose_0.pdb",
                        sequence="LISDAELEAIFEADC", ca_coords=ca)
        assert pr.ca_coords.shape == (15, 3)


class TestScoredPose:
    def test_is_pose_record(self, tmp_path: Path) -> None:
        ca = np.zeros((3, 3), dtype=np.float64)
        sp = ScoredPose(pose_idx=0, pdb_path=tmp_path / "p.pdb",
                        sequence="LIS", ca_coords=ca)
        assert isinstance(sp, PoseRecord)
        assert sp.vina_score is None
        assert sp.is_ad4_anomaly is False
```

Pattern notes:
- `tmp_path` fixture is pytest-builtin — no imports needed.
- Tests are in `tests/test_models.py`, not co-located with source (CLAUDE.md §4).
- The slow `@pytest.mark.slow` decorator is not needed here — model construction is fast.
